5/// MarkMap is a key value data structure that uses an API similar to that of
6/// `std::collections::HashMap` and `std::collections::BTreeMap`.
7///
8/// The current implementation is based on a sorted `Vec` is not optimized for
9/// storing large number of items.
10///
11/// On top of the familiar `std::collections` API it has a few additions:
12///
13/// * Marking a key with a condition.
14/// * Marking a key with a merge behavior.
15/// * A merge operator.
16///
17/// The current *conditions* are [`Used`](MarkMap::mark_used) and [`Required`](MarkMap::mark_required).
18///
19/// The available merge behaviors are [`Theirs`](MergeBehavior::Theirs), [`Ours`](MergeBehavior::Ours),
20/// and [`MutuallyExclusive`](MergeBehavior::MutuallyExclusive).
21///
22/// Merge behaviors configure how the merge operator handles cases where both `MarkMaps` being merged
23/// contain a particular key.
24#[derive(Debug, PartialEq, Eq)]
25#[doc(hidden)]
26pub struct MarkMap<K, V> {
27    default_merge_behavior: MergeBehavior,
28    contents: Vec<(K, u16, Option<V>)>,
29}
30
31/// Defines the merge behavior for a single key in the markmap.
32#[repr(u8)]
33#[derive(Clone, Copy, Eq, PartialEq, Debug)]
34#[doc(hidden)]
35pub enum MergeBehavior {
36    /// The value in `self` takes precedence.
37    Theirs = 1 << 0,
38    /// The value in `other` takes precedence.
39    Ours = 1 << 1,
40    /// It is an error if the key is present in both `MarkMaps`.
41    /// This is the default if no merge behavior is set.
42    MutuallyExclusive = 1 << 2,
43}
44
45/// Conflicting values were present while merging, and the
46/// merge behavior did not specify a means to resolve them.
47#[derive(Debug)]
48#[doc(hidden)]
49pub enum MergeError<K, V> {
50    // Contains the key which was present in both, and the value which was present in `Theirs`.
51    Exclusivity(K, V),
52}
53
54impl<K: fmt::Debug, V: fmt::Debug> Error for MergeError<K, V> {}
55
56impl<K: fmt::Debug, V> fmt::Display for MergeError<K, V> {
57    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
58        match self {
59            Self::Exclusivity(k, _) => {
60                write!(f, "Merge behavior forbids overriding key: {:?}", k)
61            }
62        }
63    }
64}

66/// A view into a single entry in a `MarkMap`, which may either be vacant or occupied.
67#[doc(hidden)]
68pub enum Entry<'a, K, V> {
69    Occupied(OccupiedEntry<'a, K, V>),
70    Vacant(VacantEntry<'a, K, V>),
71}
72
73#[repr(u8)]
74#[derive(Clone, Copy)]
75#[doc(hidden)]
76enum Mark {
77    // There are some other interesting marks that could be added based on row polymorphic records.
78    // Marks such as `Prohibited` could be used to prove disjointedness.
79    Used,
80    Required,
81    MergeBehavior(MergeBehavior),
82}
83
84impl Mark {
85    fn repr(self) -> u16 {
86        match self {
87            Self::Used => 1 << 0,
88            Self::Required => 1 << 1,
89            Self::MergeBehavior(mb) => (1 << 2) | ((mb as u16) << 8),
90        }
91    }
92}

94impl<'a, K: Ord + Clone, V> VacantEntry<'a, K, V> {
95    /// Inserts a value into a vacant entry.
96    /// Returns a mutable reference to value inserted.
97    pub fn insert(self, val: V) -> &'a mut V {
98        match self.pos {
99            Ok(pos) => {
100                self.map.contents[pos].2 = Some(val);
101                self.map.contents[pos].2.as_mut().unwrap()
102            }
103            Err(pos) => {
104                self.map.contents.insert(pos, (self.key, 0, Some(val)));
105                self.map.contents[pos].2.as_mut().unwrap()
106            }
107        }
108    }
109
110    /// Inserts a value into a vacant entry.
111    /// Returns an occupied entry.
112    pub fn insert_entry(self, val: V) -> OccupiedEntry<'a, K, V> {
113        match self.pos {
114            Ok(pos) => {
115                self.map.contents[pos].2 = Some(val);
116                OccupiedEntry { pos, map: self.map }
117            }
118            Err(pos) => {
119                self.map.contents.insert(pos, (self.key, 0, Some(val)));
120                OccupiedEntry { pos, map: self.map }
121            }
122        }
123    }
124
125    /// Marks the key associated with this entry as required.
126    pub fn mark_required(&mut self) {
127        match self.pos {
128            Ok(pos) => {
129                let repr = self.map.contents[pos].1;
130                self.map.contents[pos].1 = repr | Mark::Required.repr();
131            }
132            Err(pos) => {
133                self.map
134                    .contents
135                    .insert(pos, (self.key.clone(), Mark::Required.repr(), None));
136                self.pos = Ok(pos)
137            }
138        }
139    }
140
141    /// Sets the merge behavior for the key associated with this entry.
142    pub fn set_merge_behavior(&mut self, mb: MergeBehavior) {
143        match self.pos {
144            Ok(pos) => {
145                let merge_reprs = Mark::MergeBehavior(MergeBehavior::MutuallyExclusive).repr()
146                    | Mark::MergeBehavior(MergeBehavior::Ours).repr()
147                    | Mark::MergeBehavior(MergeBehavior::Theirs).repr();
148                let mut repr = self.map.contents[pos].1;
149                // Zap just the MergeBehavior bits.
150                repr ^= repr & merge_reprs;
151                repr |= Mark::MergeBehavior(mb).repr();
152                self.map.contents[pos].1 = repr;
153            }
154            Err(pos) => {
155                self.map.contents.insert(
156                    pos,
157                    (self.key.clone(), Mark::MergeBehavior(mb).repr(), None),
158                );
159                self.pos = Ok(pos)
160            }
161        }
162    }
163
164    /// Returns the key associated with this entry.
165    pub fn key(&self) -> &K {
166        &self.key
167    }
168}

251    /// Inserts a `key` `value` pair.
252    pub fn insert(&mut self, key: K, val: V) -> Option<V> {
253        let pos = self.contents.binary_search_by(|(k, _, _)| k.cmp(&key));
254        match pos {
255            Ok(pos) => {
256                let ret = self.contents[pos].2.take();
257                self.contents[pos].2 = Some(val);
258                ret
259            }
260            Err(pos) => {
261                self.contents.insert(pos, (key, 0, Some(val)));
262                None
263            }
264        }

377    /// Returns an `Entry` for `key`.
378    pub fn entry(&mut self, key: K) -> Entry<'_, K, V> {
379        let pos = self.contents.binary_search_by(|(k, _, _)| k.cmp(&key));
380        match pos {
381            Err(pos) => Entry::Vacant(VacantEntry {
382                pos: Err(pos),
383                key,
384                map: self,
385            }),
386            Ok(pos) => {
387                if self.contents[pos].2.is_some() {
388                    Entry::Occupied(OccupiedEntry { pos, map: self })
389                } else {
390                    Entry::Vacant(VacantEntry {
391                        pos: Ok(pos),
392                        key,
393                        map: self,
394                    })
395                }
396            }
397        }
398    }

400    /// Merges items from `other` into `self`, for each value it checks the `MergeBehavior` of self.
401    /// The `MergeBehavior` of other is not considered.
402    ///
403    /// * `MergeBehavior::Ours`: If a value is set in `other`, a value set in self takes precedence.
404    /// * `MergeBehavior::Theirs`: if a value in `other` `is_some()`, then it will overwrite values in self.
405    /// * `MergeBehavior::MutuallyExclusive` If a value is set in `other`, that value should be `None` in self.
406    /// * If `MergeBehavior` is unset, defaults to `MergeBehavior::MutuallyExclusive`.
407    ///
408    /// For the behavior of exclusive or mark the behavior as also `Mark::Required`, then after merge call `missing()`
409    /// to check all required values.
410    pub fn merge_from<U>(&mut self, mut other: MarkMap<K, U>) -> Result<(), MergeError<K, Box<V>>>
411    where
412        U: Into<V>,
413    {
414        for (their_key, their_mark, their_val) in other.contents.drain(..) {
415            let pos = self.contents.binary_search_by(|x| x.0.cmp(&their_key));
416            match pos {
417                Ok(pos) => {
418                    let (_, my_mark, my_val) = &mut self.contents[pos];
419                    let theirs_mark = Mark::MergeBehavior(MergeBehavior::Theirs).repr();
420                    let ours_mark = Mark::MergeBehavior(MergeBehavior::Ours).repr();
421                    let exclusive_mark =
422                        Mark::MergeBehavior(MergeBehavior::MutuallyExclusive).repr();
423                    let mut merge_behavior = (*my_mark & exclusive_mark)
424                        | (*my_mark & ours_mark)
425                        | (*my_mark & theirs_mark);
426                    if merge_behavior == 0 {
427                        merge_behavior = Mark::MergeBehavior(self.default_merge_behavior).repr();
428                    }
429                    if merge_behavior == exclusive_mark {
430                        // If only clippy could convince me and the borrow checker this is actually unnecessary.
431                        #[allow(clippy::unnecessary_unwrap)]
432                        if my_val.is_some() && their_val.is_some() {
433                            return Err(MergeError::Exclusivity(
434                                their_key,
435                                Box::new(their_val.unwrap().into()),
436                            ));
437                        } else if my_val.is_none() {
438                            *my_val = their_val.map(|u| u.into());
439                        }
440                    } else if merge_behavior == theirs_mark && their_val.is_some()
441                        || merge_behavior == ours_mark && my_val.is_none()
442                    {
443                        *my_mark = their_mark;
444                        *my_val = their_val.map(|u| u.into());
445                    }
446                }
447                Err(pos) => {
448                    self.contents
449                        .insert(pos, (their_key, their_mark, their_val.map(|u| u.into())));
450                }
451            }
452        }
453        Ok(())
454    }

456    /// Returns whether `key` has been marked as used.
457    pub fn is_used<Q>(&self, key: &Q) -> bool
458    where
459        K: Borrow<Q>,
460        Q: Ord + ?Sized,
461    {
462        if let Ok(pos) = self.contents.binary_search_by(|x| {
463            let k: &Q = x.0.borrow();
464            k.borrow().cmp(key)
465        }) {
466            self.contents[pos].1 & Mark::Used.repr() != 0
467        } else {
468            false
469        }
470    }

472    /// Returns a `Vec` containing all the keys that are not marked as used.
473    pub fn unused(&self) -> Vec<K> {
474        let mut ret = Vec::new();
475        for (k, mark, v) in &self.contents {
476            let used_mark = Mark::Used.repr();
477            if v.is_some() && mark & used_mark == 0 {
478                ret.push(k.to_owned())
479            }
480        }
481        ret
482    }
483
484    /// Returns a `Vec` containing all the keys that are marked as required,
485    /// but have values that are not present in the `MarkMap`.
486    pub fn missing(&self) -> Vec<&K> {
487        let mut ret = Vec::new();
488        for (k, mark, v) in &self.contents {
489            let required_mark = Mark::Required.repr();
490            if v.is_none() && mark & required_mark != 0 {
491                ret.push(k)
492            }
493        }
494        ret
495    }

509#[doc(hidden)]
510impl<K, V> Iterator for MarkMapIter<K, V> {
511    type Item = (K, V);
512
513    fn next(&mut self) -> Option<Self::Item> {
514        if !self.map.contents.is_empty() {
515            let (k, _, v) = self.map.contents.swap_remove(0);
516            v.map(|v| (k, v))
517        } else {
518            None
519        }
520    }

553/// Iterator over references to keys and values of a `MarkMap`.
554#[doc(hidden)]
555pub struct Keys<'a, K, V> {
556    pos: usize,
557    map: &'a MarkMap<K, V>,
558}
559
560impl<'a, K, V> Iterator for Keys<'a, K, V> {
561    type Item = &'a K;
562
563    fn next(&mut self) -> Option<Self::Item> {
564        loop {
565            if self.pos >= self.map.contents.len() {
566                return None;
567            }
568            let pos = self.pos;
569            self.pos += 1;
570            if self.map.contents[pos].2.is_some() {
571                return Some(&self.map.contents[pos].0);
572            }
573        }
574    }
