19#[derive(Clone, Debug, PartialEq)]
20#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
21pub struct Itemset<StorageT: Eq + Hash> {
22    pub items: HashMap<(PIdx<StorageT>, SIdx<StorageT>), Ctx, BuildHasherDefault<FnvHasher>>,
23}

25impl<StorageT: 'static + Hash + PrimInt + Unsigned> Itemset<StorageT>
26where
27    usize: AsPrimitive<StorageT>,
28{
29    /// Create a blank Itemset.
30    pub fn new(_: &YaccGrammar<StorageT>) -> Self {
31        Itemset {
32            items: HashMap::with_hasher(BuildHasherDefault::<FnvHasher>::default()),
33        }
34    }
35
36    /// Add an item `(prod, dot)` with context `ctx` to this itemset. Returns true if this led to
37    /// any changes in the itemset.
38    pub fn add(&mut self, pidx: PIdx<StorageT>, dot: SIdx<StorageT>, ctx: &Ctx) -> bool {
39        let entry = self.items.entry((pidx, dot));
40        match entry {
41            Entry::Occupied(mut e) => e.get_mut().or(ctx),
42            Entry::Vacant(e) => {
43                e.insert(ctx.clone());
44                true
45            }
46        }
47    }
48
49    /// Create a new itemset which is a closed version of `self`.
50    pub fn close(&self, grm: &YaccGrammar<StorageT>, firsts: &YaccFirsts<StorageT>) -> Self {
51        // This function can be seen as a merger of getClosure and getContext from Chen's
52        // dissertation.
53
54        let mut new_is = self.clone(); // The new itemset we're building up
55
56        // In a typical description of this algorithm, one would have a todo set which contains
57        // pairs (pidx, dot). Unfortunately this is a slow way of doing things. Searching the set
58        // for the next item and removing it is slow; and, since we don't know how many potential
59        // dots there are in a production, the set is of potentially unbounded size, so we can end
60        // up resizing memory. Since this function is the most expensive in the table generation,
61        // using a HashSet (which is the "obvious" solution) is slow.
62        //
63        // However, we can reduce these costs through two observations:
64        //   1) The initial todo set is populated with (pidx, dot) pairs that all come from
65        //      self.items.keys(). There's no point copying these into a todo list.
66        //   2) All subsequent todo items are of the form (prod_off, 0). Since the dot in these
67        //      cases is always 0, we don't need to store pairs: simply knowing which prod_off's we
68        //      need to look at is sufficient. We can represent these with a fixed-size bitfield.
69        // All we need to do is first iterate through the items in 1 and, when it's exhausted,
70        // continually iterate over the bitfield from 2 until no new items have been added.
71
72        let mut keys_iter = self.items.keys(); // The initial todo list
73        let mut zero_todos = Vob::from_elem(false, usize::from(grm.prods_len())); // Subsequent todos
74        let mut new_ctx = Vob::from_elem(false, usize::from(grm.tokens_len()));
75        loop {
76            let pidx;
77            let dot;
78            // Find the next todo item or, if there are none, break the loop. First of all we try
79            // pumping keys_iter for its next value. If there is none (i.e. we've exhausted that
80            // part of the todo set), we iterate over zero_todos.
81            match keys_iter.next() {
82                Some(&(x, y)) => {
83                    pidx = x;
84                    dot = y;
85                }
86                None => {
87                    match zero_todos.iter_set_bits(..).next() {
88                        Some(i) => {
89                            // Since zero_todos.len() == grm.prods_len, the call to as_ is safe.
90                            pidx = PIdx(i.as_())
91                        }
92                        None => break,
93                    }
94                    dot = SIdx(StorageT::zero());
95                    zero_todos.set(pidx.into(), false);
96                }
97            }
98            let prod = grm.prod(pidx);
99            if dot == grm.prod_len(pidx) {
100                continue;
101            }
102            if let Symbol::Rule(s_ridx) = prod[usize::from(dot)] {
103                // This if statement is, in essence, a fast version of what's called getContext in
104                // Chen's dissertation, folding in getTHeads at the same time. The particular
105                // formulation here is based as much on
106                // http://binarysculpting.com/2012/02/04/computing-lr1-closure/ as it is any of the
107                // several versions of getTHeads in Chen's dissertation. Nevertheless, the mapping
108                // between the two different formulations is fairly straight-forward.
109                new_ctx.set_all(false);
110                let mut nullable = true;
111                for sym in prod.iter().skip(usize::from(dot) + 1) {
112                    match *sym {
113                        Symbol::Token(s_tidx) => {
114                            new_ctx.set(usize::from(s_tidx), true);
115                            nullable = false;
116                            break;
117                        }
118                        Symbol::Rule(s_ridx) => {
119                            new_ctx.or(firsts.firsts(s_ridx));
120                            if !firsts.is_epsilon_set(s_ridx) {
121                                nullable = false;
122                                break;
123                            }
124                        }
125                    }
126                }
127                if nullable {
128                    new_ctx.or(&new_is.items[&(pidx, dot)]);
129                }
130
131                for ref_pidx in grm.rule_to_prods(s_ridx).iter() {
132                    if new_is.add(*ref_pidx, SIdx(StorageT::zero()), &new_ctx) {
133                        zero_todos.set(usize::from(*ref_pidx), true);
134                    }
135                }
136            }
137        }
138        new_is
139    }

141    /// Create a new Itemset based on calculating the goto of 'sym' on the current Itemset.
142    pub fn goto(&self, grm: &YaccGrammar<StorageT>, sym: &Symbol<StorageT>) -> Self {
143        // This is called 'transition' in Chen's dissertation, though note that the definition
144        // therein appears to get the dot in the input/output the wrong way around.
145        let mut newis = Itemset::new(grm);
146        for (&(pidx, dot), ctx) in &self.items {
147            let prod = grm.prod(pidx);
148            if dot == grm.prod_len(pidx) {
149                continue;
150            }
151            if sym == &prod[usize::from(dot)] {
152                newis.add(pidx, SIdx(dot.as_storaget() + StorageT::one()), ctx);
153            }
154        }
155        newis
156    }

172    fn test_dragon_grammar() {
173        // From http://binarysculpting.com/2012/02/04/computing-lr1-closure/
174        let grm = YaccGrammar::new(
175            YaccKind::Original(YaccOriginalActionKind::GenericParseTree),
176            "
177          %start S
178          %%
179          S: L '=' R | R;
180          L: '*' R | 'id';
181          R: L;
182          "
183        ).unwrap();
184        let firsts = grm.firsts();
185
186        let mut is = Itemset::new(&grm);
187        let mut la = Vob::from_elem(false, usize::from(grm.tokens_len()));
188        la.set(usize::from(grm.eof_token_idx()), true);
189        is.add(grm.rule_to_prods(grm.rule_idx("^").unwrap())[0], SIdx(0), &la);
190        let cls_is = is.close(&grm, &firsts);
191        println!("{:?}", cls_is);
192        assert_eq!(cls_is.items.len(), 6);

194        state_exists(&grm, &cls_is, "S", 0, SIdx(0), vec!["$"]);
