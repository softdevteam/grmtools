22/// Implements a `pushln` convenience function to `String`
23trait PushLine {
24    /// Appends a given string slice and a newline to the end of this `String`.
25    fn pushln<S: AsRef<str>>(&mut self, s: S);
26}

28impl PushLine for String {
29    fn pushln<S: AsRef<str>>(&mut self, s: S) {
30        self.push_str(s.as_ref());
31        self.push('\n');
32    }

35fn pidx_prods_data<StorageT>(ast: &GrammarAST, pidx: PIdx<StorageT>) -> (Vec<String>, Vec<Span>)
36where
37    usize: num_traits::AsPrimitive<StorageT>,
38    StorageT: 'static + num_traits::PrimInt + num_traits::Unsigned,
39{
40    let prod = &ast.prods[usize::from(pidx)];
41    prod.symbols
42        .iter()
43        .map(|sym| match sym {
44            Symbol::Rule(name, span) => (format!("'{}'", name), span),
45            Symbol::Token(name, span) => (format!("'{}'", name), span),
46        })
47        .unzip()
48}

68    pub fn ordinal(v: usize) -> String {
69        // I didn't feel like thinking about these special cases,
70        // so I just borrowed this from rusts diagnostics code.
71        let suffix = match ((11..=13).contains(&(v % 100)), v % 10) {
72            (false, 1) => "st",
73            (false, 2) => "nd",
74            (false, 3) => "rd",
75            _ => "th",
76        };
77        format!("{v}{suffix}")
78    }

80    // If a span is given returns "msg at path/file.foo:5:6" otherwise returns "msg in path/file.foo"
81    pub fn file_location_msg(&self, msg: &str, span: Option<Span>) -> String {
82        if let Some(span) = span {
83            let (line, col) = self
84                .nlc()
85                .byte_to_line_num_and_col_num(self.src, span.start())
86                .unwrap_or((0, 0));
87            format!("{} at {}:{line}:{col}", msg, self.path.display())
88        } else {
89            format!("{} in {}", msg, self.path.display())
90        }
91    }

99    // This prints a span with an optional prefix under the line numbers, which
100    // is intended to be used for denoting when lines are non-contiguous.
101    pub fn prefixed_underline_span_with_text(
102        &self,
103        prefix: &str,
104        mut span: Span,
105        s: String,
106        underline_c: char,
107    ) -> String {
108        let mut out = String::new();
109        let (start_byte, end_byte) = self.nlc().span_line_bytes(span);
110        // Produce an underline underneath a span which may cover multiple lines, and a message on the last line.
111        let mut source_lines = self.src[start_byte..end_byte].lines().peekable();
112        while let Some(source_line) = source_lines.next() {
113            let (line_start_byte, _) = self.nlc().span_line_bytes(span);
114            let span_offset_from_start = span.start() - line_start_byte;
115
116            // An underline bounded by the current line.
117            let underline_span = Span::new(
118                span.start(),
119                span.end()
120                    .min(span.start() + (source_line.len() - span_offset_from_start)),
121            );
122            let (line_num, _) = self
123                .nlc()
124                .byte_to_line_num_and_col_num(self.src, span.start())
125                .expect("Span must correlate to a line in source");
126            // Print the line_num/source text for the line.
127            out.push_str(&format!("{}| {}\n", line_num, source_line));
128            let line_num_digits = line_num.to_string().len();
129            assert!(prefix.len() <= "0| ".len());
130            // Add indentation from the start of the underlined span
131            out.push_str(&format!(
132                "{prefix}{}",
133                &" ".repeat(
134                    UnicodeWidthStr::width(&self.src[line_start_byte..underline_span.start()])
135                        + (line_num_digits + "| ".len() - prefix.len())
136                )
137            ));
138            // Add underline upto the end of line.
139            out.push_str(
140                &(underline_c.to_string()).repeat(
141                    UnicodeWidthStr::width(&self.src[underline_span.start()..underline_span.end()])
142                        .max(1),
143                ),
144            );
145
146            if source_lines.peek().is_none() {
147                // If we're at the end print the message.
148                out.push_str(&format!(" {}", &s));
149            } else {
150                // Otherwise set next span to start at the beginning of the next line.
151                out.push('\n');
152                span = Span::new(line_start_byte + source_line.len() + 1, span.end())
153            }
154        }
155
156        out
157    }

159    fn format_spanned(&self, e: &impl Spanned) -> String {
160        let mut out = String::new();
161        let mut spans = e.spans().iter().enumerate().peekable();
162        while let Some((span_num, span)) = spans.next() {
163            let (line, _) = self
164                .nlc()
165                .byte_to_line_num_and_col_num(self.src, span.start())
166                .unwrap_or((0, 0));
167            let next_line = spans
168                .peek()
169                .map(|(_, span)| span)
170                .map(|s| self.nlc().byte_to_line_num(s.start()).unwrap_or(line))
171                .unwrap_or(line);
172            // Is this line contiguous with the next, if not prefix it with dots.
173            let dots = if next_line - line > 1 { "..." } else { "" };
174            if span_num == 0 {
175                let s = e.to_string();
176                out.push_str(&self.prefixed_underline_span_with_text(dots, *span, s, '^'));
177            } else {
178                out.push('\n');
179                let s = match e.spanskind() {
180                    SpansKind::DuplicationError => {
181                        format!("{} occurrence", Self::ordinal(span_num + 1))
182                    }
183                    SpansKind::Error => {
184                        unreachable!("Should contain a single span at the site of the error")
185                    }
186                    _ => "Unrecognized spanskind".to_string(),
187                };
188                out.push_str(&self.prefixed_underline_span_with_text(dots, *span, s, '^'));
189            }
190        }
191        out
192    }

194    /// This function panics if the items in the `spans` iterator span more than a single line.
195    ///
196    /// This function only produces if all the following are true:
197    /// 1. Spans are sorted in left-to-right order.
198    /// 2. Spans are non-overlapping.
199    fn underline_spans_on_line_with_text(
200        &self,
201        spans: impl Iterator<Item = Span> + Clone,
202        msg: String,
203        underline_c: char,
204    ) -> String {
205        let mut lines = spans
206            .clone()
207            .map(|span| self.nlc().span_line_bytes(span))
208            .collect::<Vec<_>>();
209        lines.dedup();
210        assert!(lines.len() == 1);
211        let mut spans = spans.peekable();
212        let first_span = spans.peek().unwrap();
213        let mut out = String::new();
214        // For instance we want to underline the b's in
215        // aaaaa bbbb cccc bbb
216        //       ____      ___
217        // indent    indent
218        let (line_at_start, _) = self
219            .nlc()
220            .byte_to_line_num_and_col_num(self.src, first_span.start())
221            .expect("Span should correlate to a line in source");
222        let (line_start_byte, end_byte) = self.nlc().span_line_bytes(*first_span);
223        // Print the line_num/source text for the line.
224        out.push_str(&format!(
225            "{}| {}\n",
226            line_at_start,
227            &self.src[line_start_byte..end_byte]
228        ));
229        // Indent from the beginning of the line up to span.start() of the first span.
230        let line_num_digits = line_at_start.to_string().len();
231        out.push_str(&" ".repeat(
232            UnicodeWidthStr::width(&self.src[line_start_byte..first_span.start()])
233                + (line_num_digits + "| ".len()),
234        ));
235        // Draw underlines between start()..end() and indent up to the the start of the next span.
236        while let Some(span) = spans.next() {
237            out.push_str(
238                &underline_c
239                    .to_string()
240                    .repeat(UnicodeWidthStr::width(&self.src[span.start()..span.end()]).max(1)),
241            );
242            if let Some(next_span) = spans.peek() {
243                out.push_str(&" ".repeat(UnicodeWidthStr::width(
244                    &self.src[span.end()..next_span.start()],
245                )));
246            }
247        }
248        // With all the underlines drawn append the message.
249        out.push_str(&format!(" {msg}"));
250        out
251    }

253    pub fn format_conflicts<LexerTypesT>(
254        &self,
255        grm: &YaccGrammar<LexerTypesT::StorageT>,
256        ast: &GrammarAST,
257        c: &Conflicts<LexerTypesT::StorageT>,
258        _sgraph: &lrtable::StateGraph<LexerTypesT::StorageT>,
259        _stable: &lrtable::StateTable<LexerTypesT::StorageT>,
260    ) -> String
261    where
262        LexerTypesT: LexerTypes,
263        usize: num_traits::AsPrimitive<LexerTypesT::StorageT>,
264    {
265        let mut out = String::new();
266        for (t_idx, r1_prod_idx, r2_prod_idx, _st_idx) in c.rr_conflicts() {
267            let (_r1_prod_names, _r1_prod_spans) = pidx_prods_data(ast, *r1_prod_idx);
268            let (_r2_prod_names, _r2_prod_spans) = pidx_prods_data(ast, *r2_prod_idx);
269
270            let (t_name, t_span) = if let Some(name) = grm.token_name(*t_idx) {
271                (name, grm.token_span(*t_idx))
272            } else {
273                (
274                    "$",
275                    grm.token_idx("$").and_then(|t_idx| grm.token_span(t_idx)),
276                )
277            };
278            let r1_rule_idx = grm.prod_to_rule(*r1_prod_idx);
279            let r2_rule_idx = grm.prod_to_rule(*r2_prod_idx);
280            let r1_span = grm.rule_name_span(r1_rule_idx);
281            let r2_span = grm.rule_name_span(r2_rule_idx);
282            let r1_name = grm.rule_name_str(r1_rule_idx);
283            let r2_name = grm.rule_name_str(r2_rule_idx);
284
285            out.pushln(
286                self.file_location_msg(
287                    format!(
288                        "Reduce/Reduce conflict, can reduce '{}' or '{}' with lookahead '{}'",
289                        r1_name, r2_name, t_name,
290                    )
291                    .as_str(),
292                    Some(r1_span),
293                ),
294            );
295            out.pushln(self.underline_span_with_text(r1_span, "First reduce".to_string(), '^'));
296            out.pushln(self.underline_span_with_text(r2_span, "Second reduce".to_string(), '^'));
297            if let Some(t_span) = t_span {
298                out.pushln(self.underline_span_with_text(t_span, "Lookahead".to_string(), '^'));
299            }
300            out.push('\n');
301        }
302        for (s_tok_idx, r_prod_idx, _st_idx) in c.sr_conflicts() {
303            let r_rule_idx = grm.prod_to_rule(*r_prod_idx);
304            let r_rule_span = grm.rule_name_span(r_rule_idx);
305            let s_tok_span = grm.token_span(*s_tok_idx).unwrap();
306            let shift_name = grm.token_name(*s_tok_idx).unwrap();
307            let reduce_name = grm.rule_name_str(r_rule_idx);
308            let (_r_prod_names, mut r_prod_spans) = pidx_prods_data(ast, *r_prod_idx);
309            let fallback_span = ast.prods[usize::from(*r_prod_idx)].prod_span;
310            out.pushln(
311                self.file_location_msg(
312                    format!(
313                        "Shift/Reduce conflict, can shift '{}' or reduce '{}'",
314                        shift_name, reduce_name
315                    )
316                    .as_str(),
317                    Some(r_rule_span),
318                ),
319            );
320
321            out.pushln(self.underline_span_with_text(s_tok_span, "Shift".to_string(), '^'));
322            out.pushln(self.underline_span_with_text(r_rule_span, "Reduced rule".to_string(), '^'));
323
324            if r_prod_spans.is_empty() {
325                r_prod_spans.push(fallback_span);
326            }
327
328            let mut prod_lines = r_prod_spans
329                .iter()
330                .map(|span| self.nlc().span_line_bytes(*span))
331                .collect::<Vec<_>>();
332            prod_lines.sort();
333            prod_lines.dedup();
334
335            for lines in &prod_lines {
336                let mut spans_on_line = Vec::new();
337                for span in &r_prod_spans {
338                    if lines == &self.nlc().span_line_bytes(*span) {
339                        spans_on_line.push(*span)
340                    }
341                }
342
343                let msg = if Some(lines) == prod_lines.last() {
344                    "Reduced productions"
345                } else {
346                    ""
347                };
348                // We don't actually enforce left-to-right or non-overlapping,
349                // It appears that the spans always have these properties.
350                // but we lack an `Ord` impl to easily ensure it.
351                //
352                // We could convert to tuples, then back to spans or add the impl
353                out.pushln(self.underline_spans_on_line_with_text(
354                    spans_on_line.iter().copied(),
355                    msg.to_string(),
356                    '-',
357                ));
358            }
359            out.push('\n');
360        }
361        out
362    }

365pub trait DiagnosticFormatter {
366    fn format_error<E>(&self, e: E) -> Box<dyn Error>
367    where
368        E: Spanned + Error;
369    fn format_warning<W>(&self, e: W) -> String
370    where
371        W: Spanned + Display;
372}
373
374impl DiagnosticFormatter for SpannedDiagnosticFormatter<'_> {
375    fn format_error<E>(&self, e: E) -> Box<dyn Error>
376    where
377        E: Spanned + Error,
378    {
379        self.format_spanned(&e).into()
380    }
381
382    fn format_warning<W>(&self, w: W) -> String
383    where
384        W: Spanned + Display,
385    {
386        self.format_spanned(&w)
387    }
388}

648                crab.len() * 2 + lobster.len() * 2,
649            ),
650        ];
651        let spans = spans
652            .iter()
653            .map(|(start, end): &(usize, usize)| Span::new(*start, *end));
654        out.push_str(&formatter.underline_spans_on_line_with_text(
655            spans.clone(),
656            "Not crabs".to_string(),
657            '^',
658        ));
659        assert_eq!(
660            out,
661            r"
6621| 🦀🦞🦀🦞
663     ^^  ^^ Not crabs"
664        );
665    }
